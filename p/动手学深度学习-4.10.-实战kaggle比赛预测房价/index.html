import hashlib
import os
import requests

#@save
DATA_HUB = dict()  # è®°å½•æ•°æ®åˆ«å -> (ä¸‹è½½URL, æ–‡ä»¶SHA1æ ¡éªŒå€¼)
DATA_URL = 'http://d2l-data.s3-accelerate.amazonaws.com/'

def download(name, cache_dir=os.path.join('..', 'data')):  #@save
    """ä» DATA_HUB ä¸‹è½½ name å¯¹åº”æ–‡ä»¶ï¼Œè‹¥æœ¬åœ°å·²æœ‰ä¸”æ ¡éªŒé€šè¿‡åˆ™ç›´æ¥å¤ç”¨ï¼Œè¿”å›æœ¬åœ°æ–‡ä»¶è·¯å¾„"""
    assert name in DATA_HUB, f"{name} ä¸å­˜åœ¨äº DATA_HUB"
    url, sha1_hash = DATA_HUB[name]
    os.makedirs(cache_dir, exist_ok=True)
    fname = os.path.join(cache_dir, url.split('/')[-1])  # æœ¬åœ°ç¼“å­˜æ–‡ä»¶è·¯å¾„ï¼š../data/æ–‡ä»¶å.csv

    # å‘½ä¸­ç¼“å­˜ï¼šå­˜åœ¨åŒåæ–‡ä»¶åˆ™åšä¸€æ¬¡SHA1æ ¡éªŒï¼ŒåŒ¹é…åˆ™ç›´æ¥è¿”å›ï¼Œé¿å…é‡å¤ä¸‹è½½
    if os.path.exists(fname):
        sha1 = hashlib.sha1()
        with open(fname, 'rb') as f:
            # åˆ†å—è¯»å–è®¡ç®—å“ˆå¸Œï¼Œé¿å…ä¸€æ¬¡æ€§è¯»å…¥å†…å­˜
            while True:
                data = f.read(1048576)  # 1MB
                if not data:
                    break
                sha1.update(data)
        if sha1.hexdigest() == sha1_hash:
            return fname

    # æœªå‘½ä¸­ç¼“å­˜åˆ™ä¸‹è½½
    print(f'æ­£åœ¨ä»{url}ä¸‹è½½åˆ° {fname} ...')
    r = requests.get(url, stream=True, verify=True)
    with open(fname, 'wb') as f:
        f.write(r.content)
    return fname

# å°†è®­ç»ƒå’Œæµ‹è¯•çš„â€œç‰¹å¾â€æ‹¼æ¥åœ¨ä¸€èµ·ç»Ÿä¸€é¢„å¤„ç†ï¼š

    (train_data.iloc[:, 1:-1],  # è®­ç»ƒï¼šå»æ‰ Id å’Œ SalePrice

# æ‰¾åˆ°æ•°å€¼åˆ—ï¼ˆé object ç±»å‹ï¼‰ï¼›object ä¸€èˆ¬æ˜¯ç±»åˆ«/å­—ç¬¦ä¸²åˆ—
numeric_features = all_features.dtypes[all_features.dtypes != 'object'].index

# å¯¹æ•°å€¼åˆ—åš z-score æ ‡å‡†åŒ–ï¼šæ¯åˆ—å‡å‡å€¼ã€é™¤æ ‡å‡†å·®ï¼ˆæå‡æ”¶æ•›ã€é¿å…é‡çº²å·®å¼‚ï¼‰

all_features = pd.get_dummies(all_features, dummy_na=True, dtype=float)

train_labels = torch.tensor(train_data.SalePrice.values.reshape(-1, 1), dtype=torch.float32)

# ===== (4) æ¨¡å‹ã€æŸå¤±ã€æŒ‡æ ‡ =====
# ä½¿ç”¨çº¿æ€§å›å½’ï¼šy = w^T x + b
loss = nn.MSELoss()  # è®­ç»ƒæŸå¤±ï¼šå‡æ–¹è¯¯å·®ï¼ˆåœ¨â€œåŸä»·ç©ºé—´â€è®¡ç®—ï¼‰
in_features = train_features.shape[1]  # è¾“å…¥ç‰¹å¾ç»´åº¦ï¼ˆæ•°å€¼åˆ—+one-hot åˆ—çš„æ€»æ•°ï¼‰

def get_net():
    # ä¸€ä¸ªçº¿æ€§å±‚å³å¯ï¼›è¾“å‡ºç»´åº¦=1ï¼ˆé¢„æµ‹æˆ¿ä»·ï¼‰
    return nn.Sequential(nn.Linear(in_features, 1))

def log_rmse(net, features, labels):
    """è¯„ä¼°æŒ‡æ ‡ï¼šå¯¹æ•°ç©ºé—´ RMSEï¼ˆsqrt(MSE(log(pred), log(label))))ï¼‰
       æ³¨æ„ï¼šä¸ºäº†é˜²æ­¢ log(<=0)ï¼Œå¯¹é¢„æµ‹åšäº†ä¸‹é™æˆªæ–­åˆ° 1
    """
    clipped_preds = torch.clamp(net(features), 1, float('inf'))  # é¢„æµ‹å€¼å°äº1çš„æŠ¬åˆ°1
    rmse = torch.sqrt(loss(torch.log(clipped_preds), torch.log(labels)))
    return rmse.item()

# ===== (5) è®­ç»ƒå¾ªç¯ =====
def train(net, train_features, train_labels, test_features, test_labels,
          num_epochs, learning_rate, weight_decay, batch_size):
    """æ ‡å‡†è®­ç»ƒæµç¨‹ï¼šæŒ‰ batch è®­ç»ƒï¼Œè®°å½•æ¯ä¸ª epoch çš„ train/valid æŒ‡æ ‡"""
    train_ls, test_ls = [], []
    # ç”¨ d2l çš„ä¾¿æ·å‡½æ•°æŠŠ (X,y) æ‰“åŒ…æˆå°æ‰¹é‡è¿­ä»£å™¨
    train_iter = d2l.load_array((train_features, train_labels), batch_size)
    # Adam ä¼˜åŒ–å™¨ï¼ˆå¸¦ weight decay ç›¸å½“äº L2 æ­£åˆ™ï¼‰
    optimizer = torch.optim.Adam(net.parameters(), lr=learning_rate, weight_decay=weight_decay)
    for epoch in range(num_epochs):
        for X, y in train_iter:
            optimizer.zero_grad()       # 1) æ¸…æ¢¯åº¦
            l = loss(net(X), y)         # 2) å‰å‘ + è®¡ç®—æŸå¤±ï¼ˆåŸä»·ç©ºé—´çš„ MSEï¼‰
            l.backward()                # 3) åå‘ä¼ æ’­
            optimizer.step()            # 4) å‚æ•°æ›´æ–°
        # æ¯ä¸ª epoch ç»“æŸåï¼Œè®¡ç®—ä¸€æ¬¡æ•´è®­ç»ƒé›†çš„ log RMSE
        train_ls.append(log_rmse(net, train_features, train_labels))
        # è‹¥æä¾›äº†éªŒè¯é›†ï¼ˆæˆ–å«â€œæµ‹è¯•é›†æ ‡ç­¾â€ï¼‰ï¼Œè®¡ç®—éªŒè¯ log RMSE
        if test_labels is not None:
            test_ls.append(log_rmse(net, test_features, test_labels))
    return train_ls, test_ls

# ===== (6) K æŠ˜äº¤å‰éªŒè¯ =====
def get_k_fold_data(k, i, X, y):
    """å°†æ•°æ®æŒ‰é¡ºåºå‡åˆ†ä¸º k ä»½ï¼Œè¿”å›ï¼šç¬¬ i ä»½ä¸ºéªŒè¯é›†ï¼Œå…¶ä½™ k-1 ä»½æ‹¼ä¸ºè®­ç»ƒé›†"""
    assert k > 1
    fold_size = X.shape[0] // k  # æ¯æŠ˜å¤§å°ï¼ˆè‹¥ä¸èƒ½æ•´é™¤ï¼Œæœ«å°¾ä¼šä¸¢ N%k ä¸ªæ ·æœ¬ï¼‰
    X_train, y_train = None, None
    for j in range(k):
        idx = slice(j * fold_size, (j + 1) * fold_size)  # ç¬¬ j æŠ˜çš„è¡ŒåŒºé—´ [start, stop)
        X_part, y_part = X[idx, :], y[idx]
        if j == i:
            # ç¬¬ i æŠ˜ï¼šä½œä¸ºéªŒè¯é›†
            X_valid, y_valid = X_part, y_part
        elif X_train is None:
            # ç¬¬ä¸€å—è®­ç»ƒé›†ç›´æ¥åˆå§‹åŒ–
            X_train, y_train = X_part, y_part
        else:
            # åç»­è®­ç»ƒå—åœ¨è¡Œç»´æ‹¼æ¥
            X_train = torch.cat([X_train, X_part], 0)
            y_train = torch.cat([y_train, y_part], 0)
    return X_train, y_train, X_valid, y_valid

def k_fold(k, X_train, y_train, num_epochs, learning_rate, weight_decay, batch_size):
    """è·‘ K æ¬¡ï¼šæ¯æ¬¡å–ä¸€æŠ˜åšéªŒè¯ï¼Œå…¶ä½™åšè®­ç»ƒï¼›æ±‡æ€»æ¯æŠ˜çš„æœ€ç»ˆ train/valid log RMSE çš„å¹³å‡å€¼"""
    train_l_sum, valid_l_sum = 0, 0
    for i in range(k):
        data = get_k_fold_data(k, i, X_train, y_train)  # åˆ‡åˆ†ç¬¬ i æŠ˜
        net = get_net()  # æ¯æŠ˜ç”¨ä¸€ä¸ªå…¨æ–°æ¨¡å‹ï¼Œä¿è¯ç‹¬ç«‹æ€§
        train_ls, valid_ls = train(net, *data, num_epochs, learning_rate, weight_decay, batch_size)
        # å–æœ¬æŠ˜â€œæœ€åä¸€ä¸ª epochâ€çš„åˆ†æ•°ç´¯åŠ ï¼ˆç®€åŒ–åšæ³•ï¼›æ›´ç¨³åšæ³•æ˜¯æ—©åœ/å–æœ€ä¼˜ï¼‰
        train_l_sum += train_ls[-1]
        valid_l_sum += valid_ls[-1]
        # åªç”»ç¬¬ä¸€æŠ˜çš„æ”¶æ•›æ›²çº¿ï¼ˆè®­ç»ƒ vs éªŒè¯ï¼‰ï¼Œé¿å…é‡å¤
        if i == 0:
            d2l.plot(list(range(1, num_epochs + 1)), [train_ls, valid_ls],
                     xlabel='epoch', ylabel='rmse', xlim=[1, num_epochs],
                     legend=['train', 'valid'], yscale='log')
            plt.show()
        print(f'æŠ˜{i + 1}ï¼Œè®­ç»ƒlog rmse{float(train_ls[-1]):f}, éªŒè¯log rmse{float(valid_ls[-1]):f}')
    # è¿”å› K æŠ˜å¹³å‡
    return train_l_sum / k, valid_l_sum / k

# ===== 8) ç”¨å…¨é‡è®­ç»ƒé›†å†è®­ç»ƒå¹¶å¯¼å‡ºæäº¤æ–‡ä»¶ =====
def train_and_pred(train_features, test_features, train_labels, test_data,
                   num_epochs, lr, weight_decay, batch_size):
    """ä½¿ç”¨å…¨éƒ¨è®­ç»ƒé›†å†æ¬¡è®­ç»ƒä¸€éæ¨¡å‹ï¼ˆä¸ç•™éªŒè¯é›†ï¼‰ï¼Œå¯¹æµ‹è¯•é›†é¢„æµ‹å¹¶å¯¼å‡º submission.csv"""
    net = get_net()
    train_ls, _ = train(net, train_features, train_labels, None, None,
                        num_epochs, lr, weight_decay, batch_size)
    # ç”»è®­ç»ƒè¯¯å·®æ›²çº¿ï¼ˆlog RMSEï¼‰
    d2l.plot(np.arange(1, num_epochs + 1), [train_ls], xlabel='epoch',
             ylabel='log rmse', xlim=[1, num_epochs], yscale='log')
    print(f'è®­ç»ƒlog rmseï¼š{float(train_ls[-1]):f}')

    # åœ¨æµ‹è¯•é›†ä¸Šåšé¢„æµ‹ï¼ˆæ­¤å¤„è¾“å‡ºæ˜¯â€œåŸä»·ç©ºé—´â€çš„é¢„æµ‹ï¼‰
    preds = net(test_features).detach().numpy()

    submission = pd.concat([test_data['Id'], test_data['SalePrice']], axis=1)
    submission.to_csv('submission.csv', index=False)  # ä¿å­˜æäº¤æ–‡ä»¶

train_and_pred(train_features, test_features, train_labels, test_data,
               num_epochs, lr, weight_decay, batch_size)
