def dijkstra(graph,start_index,max):
    # ÂàùÂßãÂåñcost,path,visitedÊï∞ÁªÑ

    # costÊï∞ÁªÑÔºöcost[i]Ë°®Á§∫‰ªéstart_indexÂà∞iÂè∑ÂÖÉÁ¥†ÁöÑÊúÄÂ∞èËä±Ë¥π
    cost=[max]*len(graph)
    cost[start_index] = 0

    # pathÊï∞ÁªÑÔºöpath[i]Ë°®Á§∫‰ªéstart_indexÂà∞iÂè∑ÂÖÉÁ¥†ÁöÑÊúÄÁü≠Ë∑ØÂæÑ‰∏≠ÔºåÂà∞ËææiÂè∑ÂÖÉÁ¥†ÁöÑÂâç‰∏Ä‰∏™ÂÖÉÁ¥†Á¥¢Âºï‰∏∫path[i]ÔºàÂç≥ÊÉ≥Ë¶Å‰ª•ÊúÄÂ∞èËä±Ë¥πÂà∞ËææiÂè∑ÂÖÉÁ¥†ÔºåÈúÄË¶ÅÈÄöËøápath[i]Âè∑ÂÖÉÁ¥†Ôºâ
    path = [-1] * len(graph)
    path[start_index] = start_index

    # visitedÊï∞ÁªÑÔºövisited[i]Ë°®Á§∫iÂè∑ÂÖÉÁ¥†ÊòØÂê¶Ë¢´ËÆøÈóÆËøáÔºàÂ∑≤ÁªèÊâæÂà∞‰∫ÜÊúÄ‰ºòË∑ØÂæÑÔºâ
    visited=[0]*len(graph)
    visited[start_index]=1

    for i in range(len(graph)):
        if(visited[i]==0):
            cost[i]=graph[start_index][i]
            # path[i]=-1Ë°®Á§∫‰∏çÂèØËææ
            path[i]=(start_index if(cost[i]<max) else -1)

    # ‰∏ª‰Ωì‰ª£Á†Å
    for i in range(1,len(graph)):
        cur_index=-1
        min_cost=max
        # ÊâæÂà∞‰∏ã‰∏ÄÊ≠•‰ª£‰ª∑ÊúÄÂ∞èÁöÑËäÇÁÇπ
        for j in range(len(graph)):
            if(visited[j]==0):
                if(cost[j]<min_cost):
                    min_cost=cost[j]
                    cur_index=j

        # Â¶ÇÊûúÊ≤°ÊúâÊâæÂà∞ÂèØËÆøÈóÆÁöÑËäÇÁÇπÔºåÈÄÄÂá∫Âæ™ÁéØ
        if cur_index==-1:
            break

        # Ê†áËÆ∞‰∏ã‰∏ÄÊ≠•‰ª£‰ª∑ÊúÄÂ∞èÁöÑËäÇÁÇπ‰∏∫Â∑≤ËÆøÈóÆÁöÑËäÇÁÇπÔºàÂΩìÂâçËäÇÁÇπÔºâ
        visited[cur_index]=1

        # ‰æùÊçÆÊâæÂà∞ÁöÑ‰∏ã‰∏ÄÊ≠•‰ª£‰ª∑ÊúÄÂ∞èÁöÑËäÇÁÇπcur_indexÊõ¥Êñ∞ÂÖ∂ÈôÑËøë‰∏ÄÂúàÁöÑËäÇÁÇπÊï∞ÊçÆ
        for k in range(len(graph)):
            if(visited[k]==0):
                if(cost[cur_index]+graph[cur_index][k]<cost[k]):
                    cost[k]=cost[cur_index]+graph[cur_index][k]
                    path[k]=cur_index
    return path

def get_path(path, start_index, target_index):
    if path[target_index] == -1:

    path_list = []
